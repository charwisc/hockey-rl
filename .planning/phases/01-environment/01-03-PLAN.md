   ```python
   body = self._mjcf_root.worldbody.add('body', name='body')
   # Capsule: radius=0.3m, half-height=0.4m, mass=75kg

            size=[0.3, 0.4],
            pos=[0, 0, 0.7],
            mass=75.0,
            contype=2, conaffinity=2,   # agents collide with each other (bit 1)
            rgba=[1.0, 0.2, 0.2, 1.0] if team == 0 else [0.2, 0.2, 1.0, 1.0])
   ```

3. Add stick hitbox extending forward from player:
   ```python
   body.add('geom', name='stick', type='box',
            size=[0.1, 0.5, 0.05],
            pos=[0.4, 0, 0.3],
            mass=1.0,
            contype=1, conaffinity=1,   # bit 0: stick collides with puck (puck contype=5 has bit 0)
            rgba=[0.6, 0.4, 0.2, 1.0])
   ```

   Collision math: stick contype=1 (bit 0), puck conaffinity=5 (bits 0+2). `stick.contype & puck.conaffinity = 1 & 5 = 1` -> contact generated. Puck contype=5 (bits 0+2), board conaffinity=4 (bit 2). `puck.contype & board.conaffinity = 5 & 4 = 4` -> contact generated. Agent capsule contype=2 (bit 1), board conaffinity=4 (bit 2). `capsule.contype & board.conaffinity = 2 & 4 = 0` -> NO contact (agents don't collide with boards — joint limits enforce boundaries instead).

4. Add joints with limits to keep agents on the rink:
   ```python
   # Slide joints: x and y translation, limited to slightly inside rink
   body.add('joint', name='x', type='slide', axis=[1, 0, 0],
            limited=True, range=[-14.5, 14.5],  # rink_length/2 - 0.5
            damping=10.0)
   body.add('joint', name='y', type='slide', axis=[0, 1, 0],
            limited=True, range=[-7.0, 7.0],    # rink_width/2 - 0.5
            damping=10.0)
   # Hinge for z-rotation (facing direction)
   body.add('joint', name='rot', type='hinge', axis=[0, 0, 1],
            damping=5.0)
   ```

5. Add velocity-controlled actuators:
   ```python
   self._mjcf_root.actuator.add('velocity', name='vx',
                                 joint=body.find('joint', 'x'), kv=200.0)
   self._mjcf_root.actuator.add('velocity', name='vy',
                                 joint=body.find('joint', 'y'), kv=200.0)
   self._mjcf_root.actuator.add('velocity', name='vrot',
                                 joint=body.find('joint', 'rot'), kv=50.0)
   ```

   NOTE: The action space is 4 floats [move_x, move_y, speed, stick_angle] per agent. The mapping from 4 action floats to 3 actuators is: `vx_target = move_x * speed * MAX_SPEED`, `vy_target = move_y * speed * MAX_SPEED`, `vrot_target = stick_angle * MAX_ROT_SPEED`. This mapping is done in HockeyTask (not in the entity).

```python
"""Hockey 2v2 task: observation assembly, reward computation, episode management."""

import numpy as np
from dm_control import composer
from dm_control.utils import rewards as dmr

from env.obs_spec import OBS_SPEC, OBS_DIM

# Action mapping constants
MAX_SPEED = 5.0        # m/s max agent speed
MAX_ROT_SPEED = 3.0    # rad/s max rotation speed
POSSESSION_DIST = 0.5  # meters: stick-puck distance threshold for possession

# Face-off positions (world frame)
FACEOFF_POSITIONS = {
    # (team, player_idx): (x, y)
    (0, 0): (-5.0, 2.0),
    (0, 1): (-5.0, -2.0),
    (1, 0): (5.0, 2.0),
    (1, 1): (5.0, -2.0),
}


class HockeyTask(composer.Task):

    def __init__(self, arena, players, puck, time_limit=60.0):
        self._arena = arena
        self._players = players   # list of 4: [team0_p0, team0_p1, team1_p0, team1_p1]
        self._puck = puck
        self._time_limit = time_limit
        self._score = [0, 0]      # [team0, team1]
        self._goal_scored_this_step = False

    @property
    def root_entity(self):
        return self._arena

    def initialize_episode_mjcf(self, random_state):
        """Attach entities to arena. Called once at compile time."""
        for player in self._players:
            self._arena.attach(player)
        self._arena.attach(self._puck)

    def initialize_episode(self, physics, random_state):
        """Reset to face-off: players to starting positions, puck to center."""
        self._score = [0, 0]
        self._goal_scored_this_step = False

        # Reset puck to center with zero velocity
        puck_x = self._puck.mjcf_model.find('joint', 'puck_x')
        puck_y = self._puck.mjcf_model.find('joint', 'puck_y')
        puck_rot = self._puck.mjcf_model.find('joint', 'puck_rot')
        physics.bind(puck_x).qpos = 0.0
        physics.bind(puck_y).qpos = 0.0
        physics.bind(puck_rot).qpos = 0.0
        physics.bind(puck_x).qvel = 0.0
        physics.bind(puck_y).qvel = 0.0
        physics.bind(puck_rot).qvel = 0.0

        # Reset players to face-off positions
        for i, player in enumerate(self._players):
            team = player.team
            pidx = player.player_idx
            fx, fy = FACEOFF_POSITIONS[(team, pidx)]
            x_joint = player.mjcf_model.find('joint', 'x')
            y_joint = player.mjcf_model.find('joint', 'y')
            rot_joint = player.mjcf_model.find('joint', 'rot')
            physics.bind(x_joint).qpos = fx
            physics.bind(y_joint).qpos = fy
            physics.bind(rot_joint).qpos = 0.0
            physics.bind(x_joint).qvel = 0.0
            physics.bind(y_joint).qvel = 0.0
            physics.bind(rot_joint).qvel = 0.0

    def before_step(self, physics, action, random_state):
        """Map 12-float action array to 12 actuator controls.

        action shape: (12,) — 3 actuator targets per player x 4 players.
        But externally, each agent provides 4 floats [move_x, move_y, speed, stick_angle].
        The HockeyEnv wrapper handles the 4-float -> 3-actuator mapping and
        concatenation. Here we just apply the 12-float array directly to actuators.
        """
        self._goal_scored_this_step = False
        if action is not None and len(action) > 0:
            physics.set_control(action)

    def after_step(self, physics, random_state):
        """Check for goal scoring after physics step."""
        puck_x = physics.bind(
            self._puck.mjcf_model.find('joint', 'puck_x')).qpos
        puck_y = physics.bind(
            self._puck.mjcf_model.find('joint', 'puck_y')).qpos

        # Goal detection: puck crosses goal line within goal width
        goal_half_width = 2.0
        rink_half = self._arena.rink_length / 2

        # Home goal (team 0 defends): puck at x < -rink_half + 1.0
        if puck_x < -rink_half + 1.0 and abs(puck_y) < goal_half_width:
            self._score[1] += 1  # team 1 scores
            self._goal_scored_this_step = True

        # Away goal (team 1 defends): puck at x > rink_half - 1.0
        if puck_x > rink_half - 1.0 and abs(puck_y) < goal_half_width:
            self._score[0] += 1  # team 0 scores
            self._goal_scored_this_step = True

    def get_observation(self, physics):
        """Return empty dict — obs assembly happens in HockeyEnv wrapper."""
        return {}

    def get_reward(self, physics):
        """Single scalar reward for the dm_env TimeStep (not used directly)."""
        return 0.0

    def should_terminate_episode(self, physics):
        """End on goal or time limit."""
        return (self._goal_scored_this_step or
                physics.time() >= self._time_limit)

    # ---- Per-agent observation and reward methods (called by HockeyEnv) ----

    def build_obs_for_agent(self, physics, agent_idx: int) -> np.ndarray:
        """Build the 22-float egocentric obs vector for one agent.

        Uses OBS_SPEC slices from obs_spec.py to ensure index stability.
        """
        obs = np.zeros(OBS_DIM, dtype=np.float32)
        player = self._players[agent_idx]
        team = player.team

        # Agent's own joints
        x_joint = player.mjcf_model.find('joint', 'x')
        y_joint = player.mjcf_model.find('joint', 'y')
        rot_joint = player.mjcf_model.find('joint', 'rot')

        # obs[0:2] = agent_pos
        obs[0] = physics.bind(x_joint).qpos
        obs[1] = physics.bind(y_joint).qpos
        # obs[2:4] = agent_vel
        obs[2] = physics.bind(x_joint).qvel
        obs[3] = physics.bind(y_joint).qvel

        # Teammate: same team, other index
        teammate_idx = self._get_teammate_idx(agent_idx)
        tm = self._players[teammate_idx]
        tm_x = tm.mjcf_model.find('joint', 'x')
        tm_y = tm.mjcf_model.find('joint', 'y')
        # obs[4:6] = teammate_pos
        obs[4] = physics.bind(tm_x).qpos
        obs[5] = physics.bind(tm_y).qpos
        # obs[6:8] = teammate_vel
        obs[6] = physics.bind(tm_x).qvel
        obs[7] = physics.bind(tm_y).qvel

        # Opponents
        opp_indices = self._get_opponent_indices(agent_idx)
        for i, opp_idx in enumerate(opp_indices):
            opp = self._players[opp_idx]
            opp_x = opp.mjcf_model.find('joint', 'x')
            opp_y = opp.mjcf_model.find('joint', 'y')
            base = 8 + i * 4  # opp0: [8:12], opp1: [12:16]
            obs[base] = physics.bind(opp_x).qpos
            obs[base + 1] = physics.bind(opp_y).qpos
            obs[base + 2] = physics.bind(opp_x).qvel
            obs[base + 3] = physics.bind(opp_y).qvel

        # Puck
        puck_x = self._puck.mjcf_model.find('joint', 'puck_x')
        puck_y = self._puck.mjcf_model.find('joint', 'puck_y')
        # obs[16:18] = puck_pos
        obs[16] = physics.bind(puck_x).qpos
        obs[17] = physics.bind(puck_y).qpos
        # obs[18:20] = puck_vel
        obs[18] = physics.bind(puck_x).qvel
        obs[19] = physics.bind(puck_y).qvel

        # obs[20] = stick_angle (relative to agent facing)
        agent_facing = float(physics.bind(rot_joint).qpos)
        obs[20] = 0.0  # RESERVED: always-zero in v1.0.0. No independent stick joint exists;
        # the 4-float action's stick_angle input controls body vrot, not a separate
        # stick. Downstream consumers (ONNX export, JS mirror) must not use this
        # dimension for physics mirroring. See obs_spec.py for the canonical layout.
        # obs[21] = facing_angle
        obs[21] = agent_facing

        return obs

    def get_reward_components(self, physics, agent_idx: int) -> dict:
        """Compute all 6 reward components for one agent.

        Returns dict with keys: r_goal, r_puck_toward_goal, r_possession,
        r_positioning, r_clustering, r_step_penalty.
        """
        player = self._players[agent_idx]
        team = player.team

        # Agent position
        x_joint = player.mjcf_model.find('joint', 'x')
        y_joint = player.mjcf_model.find('joint', 'y')
        agent_pos = np.array([
            physics.bind(x_joint).qpos,
            physics.bind(y_joint).qpos
        ])

        # Puck position and velocity
        puck_x_j = self._puck.mjcf_model.find('joint', 'puck_x')
        puck_y_j = self._puck.mjcf_model.find('joint', 'puck_y')
        puck_pos = np.array([
            physics.bind(puck_x_j).qpos,
            physics.bind(puck_y_j).qpos
        ])
        puck_vel = np.array([
            physics.bind(puck_x_j).qvel,
            physics.bind(puck_y_j).qvel
        ])

        # Opponent goal position (where this agent wants to score)
        rink_half = self._arena.rink_length / 2
        if team == 0:
            opponent_goal = np.array([rink_half, 0.0])  # team 0 attacks +x
        else:
            opponent_goal = np.array([-rink_half, 0.0])  # team 1 attacks -x

        # --- r_goal: sparse +10 when scoring team matches agent's team ---
        r_goal = 0.0
        if self._goal_scored_this_step:
            if team == 0 and self._score[0] > 0:
                r_goal = 10.0
            elif team == 1 and self._score[1] > 0:
                r_goal = 10.0
            # Also penalize own-goal (opponent scores)
            if team == 0 and self._score[1] > 0:
                r_goal = -10.0
            elif team == 1 and self._score[0] > 0:
                r_goal = -10.0

        # --- Possession check: is this agent's stick close to the puck? ---
        # Approximate stick tip position: agent_pos + offset in facing direction
        agent_facing = float(physics.bind(
            player.mjcf_model.find('joint', 'rot')).qpos)
        stick_tip = agent_pos + 0.4 * np.array([
            np.cos(agent_facing), np.sin(agent_facing)])
        has_possession = np.linalg.norm(stick_tip - puck_pos) < POSSESSION_DIST

        # --- r_puck_toward_goal: gated on possession (prevents proxy exploit) ---
        goal_dir = opponent_goal - puck_pos
        goal_dist = np.linalg.norm(goal_dir)
        goal_dir = goal_dir / (goal_dist + 1e-8)
        puck_toward = float(np.dot(puck_vel, goal_dir))
        r_puck_toward_goal = np.clip(puck_toward, -1.0, 1.0) * 0.05 if has_possession else 0.0

        # --- r_possession: small bonus for being near the puck ---
        r_possession = 0.01 if has_possession else 0.0

        # --- r_positioning: distance to ideal support position ---
        # Ideal: between own goal and puck, offset toward center
        own_goal = np.array([-rink_half if team == 0 else rink_half, 0.0])
        ideal_pos = 0.5 * (own_goal + puck_pos)  # midpoint
        dist_to_ideal = float(np.linalg.norm(agent_pos - ideal_pos))
        r_positioning = float(dmr.tolerance(
            dist_to_ideal, bounds=(0, 3.0), margin=5.0)) * 0.005

        # --- r_clustering: penalize being too close to teammate ---
        teammate_idx = self._get_teammate_idx(agent_idx)
        tm = self._players[teammate_idx]
        tm_pos = np.array([
            physics.bind(tm.mjcf_model.find('joint', 'x')).qpos,
            physics.bind(tm.mjcf_model.find('joint', 'y')).qpos
        ])
        teammate_dist = float(np.linalg.norm(agent_pos - tm_pos))
        r_clustering = -0.01 if teammate_dist < 2.0 else 0.0

        # --- r_step_penalty: constant per-step cost ---
        r_step_penalty = -0.001

        return {
            'r_goal': float(r_goal),
            'r_puck_toward_goal': float(r_puck_toward_goal),
            'r_possession': float(r_possession),
            'r_positioning': float(r_positioning),
            'r_clustering': float(r_clustering),
            'r_step_penalty': float(r_step_penalty),
        }

    def get_total_reward(self, physics, agent_idx: int) -> float:
        """Sum of all reward components for one agent."""
        return sum(self.get_reward_components(physics, agent_idx).values())

    # ---- Helpers ----

    def _get_teammate_idx(self, agent_idx: int) -> int:
        """Agent 0<->1, 2<->3."""
        if agent_idx < 2:
            return 1 - agent_idx
        else:
            return 5 - agent_idx  # 2->3, 3->2

    def _get_opponent_indices(self, agent_idx: int) -> list:
        """Return indices of both opponents."""
        if agent_idx < 2:
            return [2, 3]
        else:
            return [0, 1]

    @property
    def score(self):
        return list(self._score)

    @property
    def goal_scored_this_step(self):
        return self._goal_scored_this_step
```
